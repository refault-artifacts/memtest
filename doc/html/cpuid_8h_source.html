214static inline void cpuid(uint32_t op, uint32_t count, uint32_t *eax, uint32_t *ebx, uint32_t *ecx, uint32_t *edx)
215{
216 *eax = op;
217 *ecx = count;
218 __asm__ __volatile__ ("cpuid"
219 : "=a" (*eax),
220 "=b" (*ebx),
221 "=c" (*ecx),
222 "=d" (*edx)
223 : "0" (*eax),
224 "2" (*ecx)
225 );
226}
