155{

157 char e_magic[2]; /* Magic number, 0x5a4d */

159 char e_cp[2]; /* Pages in file, 0x3 */
160 char e_crlc[2]; /* Relocations, 0x0 */
161 char e_cparhdr[2]; /* Size of header in paragraphs, 0x4 */

163 char e_maxalloc[2]; /* Maximum extra paragraphs needed, 0xFFFF */
164 char e_ss[2]; /* Initial (relative) SS value, 0x0 */
165 char e_sp[2]; /* Initial SP value, 0xb8 */
166 char e_csum[2]; /* Checksum, 0x0 */
167 char e_ip[2]; /* Initial IP value, 0x0 */
168 char e_cs[2]; /* Initial (relative) CS value, 0x0 */
169 char e_lfarlc[2]; /* File address of relocation table, 0x40 */
170 char e_ovno[2]; /* Overlay number, 0x0 */
171 char e_res[4][2]; /* Reserved words, all 0x0 */
172 char e_oemid[2]; /* OEM identifier (for e_oeminfo), 0x0 */
173 char e_oeminfo[2]; /* OEM information; e_oemid specific, 0x0 */
174 char e_res2[10][2]; /* Reserved words, all 0x0 */
175 char e_lfanew[4]; /* File address of new exe header, usually 0x80 */
176 char dos_message[16][4]; /* other stuff, always follow DOS header */
177
178 /* Note: additional bytes may be inserted before the signature. Use
179 the e_lfanew field to find the actual location of the NT signature */
180
181 char nt_signature[4]; /* required NT signature, 0x4550 */
182
183 /* From standard header */
184
185 char f_magic[2]; /* magic number */
186 char f_nscns[2]; /* number of sections */
187 char f_timdat[4]; /* time & date stamp */
188 char f_symptr[4]; /* file pointer to symtab */
189 char f_nsyms[4]; /* number of symtab entries */
190 char f_opthdr[2]; /* sizeof(optional hdr) */
191 char f_flags[2]; /* flags */
192};
