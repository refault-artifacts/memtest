159/* Returns the default maximum packet size for a USB device running at the
160 * given speed.
161 *
162 * Used internally by the various HCI drivers.
163 */
164static inline int default_max_packet_size(usb_speed_t device_speed)
165{
166 switch (device_speed) {
167 case USB_SPEED_LOW:
168 return 8;
169 case USB_SPEED_FULL:
170 return 64;
171 case USB_SPEED_HIGH:
172 return 64;
173 default:
174 return 0;
175 }
176}

184static inline bool valid_usb_max_packet_size(int size, usb_speed_t speed)
185{
186 return (size == 8) || ((speed != USB_SPEED_LOW) && (size == 16 || size == 32 || size == 64));
187}
